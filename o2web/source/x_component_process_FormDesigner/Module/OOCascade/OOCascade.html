使用脚本选项说明

返回一个可选值数组列表，如下：

return [{
    text: "商品",     //选项文本
    value: "goods",  //选项值
    children: [{     //子选项，如果没有children表示叶子节点
        text: "床上用品",
        value: "bedding",
        //子选项也可以使用动态加载的函数，字符串为该函数名。
        children: "getSubOptions"
    }]
}]

返回一个resolve了可选值数组的Promise对象，如下：

return new Promise((resolve)=>{
    window.setTimeout(()=>{
        let items = [{
            text: "商品",
            value: "goods",
            children: [{
                text: "床上用品",
                value: "bedding"
            }]
        }]
        resolve(items) //返回选项数据
    }, 500)
})

动态加载选项说明

需返回一个带名称的函数，该函数返回一个Promise，如下：

value为选项值路径，如：['商品','床上用品'，'枕头']。根节点路径为空数组[]。

e为触发事件对象，可以通过e.target获取触发的Dom节点。

return function getSubOptions(value, e){
    return new Promise((resolve)=>{
        window.setTimeout(()=>{
            let items;
            const v = value.at(-1); //返回最后一项
            if( !v ){ //空，为根节点
                items = [{
                    text: "商品", //选项文本
                    value: "goods",//选项值
                    //子选项，和返回的方法同名。表示展开该节点继续调用本函数。
                    children: getSubOptions
                }]
            }else if(v === 'goods'){
                items = [{
                    text: "床上用品",
                    value: "bedding",
                    children: [{
                        text: "枕头",
                        value: "pillow"
                    }]
                }]
            }
            resolve(items) //返回选项数据
        }, 500)
    })
}
